import io.restassured.RestAssured;
import io.restassured.response.Response;
import org.testng.Assert;
import org.testng.annotations.BeforeClass;
import org.testng.annotations.Test;

import static io.restassured.RestAssured.given;

public class CreateUserAPITest {

    private String baseUrl;

    @BeforeClass
    public void setup() {
        // Base URL fetched from Environment Variable
        baseUrl = System.getenv("BASE_URL");
        if (baseUrl == null) {
            throw new RuntimeException("BASE_URL environment variable is not set!");
        }

        RestAssured.baseURI = baseUrl;
    }

    @Test
    public void testCreateUser() {

        // JSON Request Body for User Creation API
        String requestBody = "{\n" +
                "  \"name\": \"John Doe\",\n" +
                "  \"email\": \"john.doe@example.com\",\n" +
                "  \"password\": \"Test@123\"\n" +
                "}";

        // POST request
        Response response = 
                given()
                    .header("Content-Type", "application/json")
                    .body(requestBody)
                .when()
                    .post("/users/create")   // Sample endpoint, you may replace it
                .then()
                    .extract()
                    .response();

        // Validate ONLY Status Code (As selected)
        Assert.assertEquals(response.getStatusCode(), 201, "Status code validation failed!");

        System.out.println("Response Status Code: " + response.getStatusCode());
        System.out.println("Response Body: " + response.getBody().asString());
    }
}
